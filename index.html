Assignment

Taints & Tolerants

Set taint taint-effect to NoSchedule | PreferNoSchedule | NoExecute

  • NoSchedule, Pod will not be scheduled

  • PrefereNoSchedule, Pod will be tried not to schedule

  • NoExecute, New Pods will not be scheduled, existing Pods will be evicted

xxxxxxxxxx
# Set taint
kubectl taint nodes node-name key=value:taint-effect
# List taint
kubectl describe node node-name | grep Taint


xxxxxxxxxx
kubectl taint nodes node-name app=blue:NoSchedule

Set Toleration

xxxxxxxxxx
apiVersion: v1
kind: Pod
metadata:
  name: myapp-rs
spec:
  containers:
  - name: nginx-container
    image: nginx
  tolerations:
  - key: "app"
    operator: "Equal"
    value: "blue"
    effect: "NoSchedule"

xxxxxxxxxx
apiVersion: v1
kind: Pod
metadata:
  name: myapp-rs
spec:
  containers:
  - name: nginx-container
    image: nginx
  affinity:
    nodeAffinity:
      requiredDuringSchedulingIgnoredDuringExecution:
        nodeSelectorTerms:
        - matchExpressions:
          - key: size
            operator: In | NotIn | Exists
            values:
            - Large

Build and run

xxxxxxxxxx
# build from image
docker build -t {image tag} .
# list containers
docker ps -a
# start container 
docker run {containerid}
# mapped port , detached with name and removal on stop
docker run -p 3000:80 -d --name feedback-app --rm {image-name}
# docker container commandline
docker exec -t -i {containerid} /bin/bash
# docker port, detached with name, volume and removal on stop
docker run -p 3000:80 -d --name feedback-app -v feedback:/app/feedback --rm {image-name}
# docker port, detached with name, volume and removal on stop
docker run -p 3000:80 -d --name feedback-app -v feedback:/app/feedback -v "{fq_path}":/app --rm {image-name}
## Volumes
# Named volume: -v feedback:/app/feedback
# Anonymous volumes: -v /app/node_modules
# Bind mount: -v "/Users/username/../somehere/on/local/machine":/app/feedback
# Bind mount Read Only: -v "/Users/username/../somehere/on/local/machine":/app/feedback:ro
# docker log file
docker logs --follow {container id}
